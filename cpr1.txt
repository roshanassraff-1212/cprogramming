#include <stdio.h>
int main() {
    int n, i;
    int sum =0, average;
    printf("Enter the number of people: ");
    scanf("%d", &n);
    int age[n];
    printf("Enter the age for %d people:\n", n);
    for(i = 0; i < n; i++) {
        printf("Person %d: ", i + 1);
        scanf("%d", &age[i]);
        sum+=age[i];
    }
    average = sum / n;
    printf("Total Sum of Ages: %d\n", sum);
    printf("Average Age: %d\n", average);
    return 0;
}
