#include <stdio.h>
#include <stdlib.h>

struct Node {
    int pageId;
    struct Node* prev;
    struct Node* next;
};

void insertAtEnd(struct Node** head, int pageId) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->pageId = pageId;
    newNode->next = NULL;

    if (*head == NULL) {
        newNode->prev = NULL;
        *head = newNode;
        return;
    }

    struct Node* temp = *head;
    while (temp->next != NULL)
        temp = temp->next;

    temp->next = newNode;
    newNode->prev = temp;
}

void deleteByValue(struct Node** head, int value) {
    struct Node* temp = *head;

    while (temp != NULL && temp->pageId != value)
        temp = temp->next;

    if (temp == NULL)
        return;

    if (temp->prev != NULL)
        temp->prev->next = temp->next;
    else
        *head = temp->next;

    if (temp->next != NULL)
        temp->next->prev = temp->prev;

    free(temp);
}

void display(struct Node* head) {
    struct Node* temp = head;
    while (temp != NULL) {
        printf("%d<->", temp->pageId);
        temp = temp->next;
    }
    printf("NULL");
}

int main() {
    struct Node* head = NULL;

    insertAtEnd(&head, 11);
    insertAtEnd(&head, 22);
    insertAtEnd(&head, 33);
    insertAtEnd(&head, 44);

    deleteByValue(&head, 33);

    display(head);

    return 0;
}
