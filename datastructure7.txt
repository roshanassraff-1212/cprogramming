#include <stdio.h>

/* Define Address structure */
struct Address {
    char city[50];
    int pincode;
};

/* Define Student structure with nested Address */
struct Student {
    int id;
    char name[50];
    struct Address addr;
};

int main() {
    /* Declare a Student variable */
    struct Student s1;

    /* Pointer to Student structure */
    struct Student *ptr = &s1;

    /* Input student details */
    printf("Enter Student ID: ");
    scanf("%d", &ptr->id);

    printf("Enter Student Name: ");
    scanf(" %[^\n]", ptr->name);

    printf("Enter City: ");
    scanf(" %[^\n]", ptr->addr.city);

    printf("Enter Pincode: ");
    scanf("%d", &ptr->addr.pincode);

    /* Display student details using structure pointer */
    printf("\n--- Student Details ---\n");
    printf("ID: %d\n", ptr->id);
    printf("Name: %s\n", ptr->name);
    printf("City: %s\n", ptr->addr.city);
    printf("Pincode: %d\n", ptr->addr.pincode);

    return 0;
}
